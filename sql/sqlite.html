      # SQLite С║▓тњї(Affinity)у▒╗тъІ

      SQLite Тћ»ТїЂтѕЌуџёС║▓тњїу▒╗тъІТдѓт┐хсђѓС╗╗СйЋтѕЌС╗ЇуёХтЈ»С╗ЦтГўтѓеС╗╗СйЋу▒╗тъІуџёТЋ░ТЇ«№╝їтйЊТЋ░ТЇ«ТЈњтЁЦТЌХ№╝їУ»ЦтГЌТ«хуџёТЋ░ТЇ«т░єС╝џС╝ўтЁѕжЄЄућеС║▓у╝ўу▒╗тъІСйюСИ║У»Цтђ╝уџётГўтѓеТќ╣т╝ЈсђѓSQLite уЏ«тЅЇуџёуЅѕТюгТћ»ТїЂС╗ЦСИІС║ћуДЇС║▓у╝ўу▒╗тъІ№╝џ

      INTEGER т»╣С║јС║▓у╝ўу▒╗тъІСИ║ INTEGER уџётГЌТ«х№╝їтЁХУДётѕЎуГЅтљїС║ј NUMERIC№╝їтћ»СИђти«тѕФТў»тюеТЅДУАї CAST УАеУЙЙт╝ЈТЌХсђѓ

      public class Sample {
    public static void main(String[] args) {
        Connection connection = null;
        try {
            // тѕЏт╗║ТЋ░ТЇ«т║ЊУ┐ъТјЦ
            connection = DriverManager.getConnection("jdbc:sqlite:sample.db");
            Statement statement = connection.createStatement();
            statement.setQueryTimeout(30);  // У«Йуй« sql ТЅДУАїУХЁТЌХТЌХжЌ┤СИ║ 30s

            statement.executeUpdate("drop table if exists person");
            statement.executeUpdate("create table person (id integer, name string)");
            statement.executeUpdate("insert into person values(1, 'leo')");
            statement.executeUpdate("insert into person values(2, 'yui')");
            ResultSet rs = statement.executeQuery("select * from person");
            while (rs.next()) {
                // У»╗тЈќу╗ЊТъюжЏє
                System.out.println("name = " + rs.getString("name"));
                System.out.println("id = " + rs.getInt("id"));
            }
        } catch (SQLException e) {
            // тдѓТъюжћЎУ»»С┐АТЂ»Тў» "out of memory"№╝їтЈ»УЃйТў»ТЅЙСИЇтѕ░ТЋ░ТЇ«т║ЊТќЄС╗Х
            System.err.println(e.getMessage());
        } finally {
            try {
                if (connection != null) {
                    connection.close();
                }
            } catch (SQLException e) {
                // тЁ│жЌГУ┐ъТјЦтц▒У┤Ц
                System.err.println(e.getMessage());
            }
        }
    }
}
